static void print_prompt(void) {
    // 先組出從 root 到 g_cwd 的路徑（簡化版，只顯示當前目錄名）
    const char *dirname = g_cwd->name;   // root 時是 "/"

    // 如果是 root：TONY_FS:/
    // 其他目錄：   TONY_FS:dirname*
    if (g_cwd == g_root) {
        printf("TONY_FS:/ > ");
    } else {
        printf("TONY_FS:%s* > ", dirname);
    }
}
